//
// Syntax is as follows:
//
//		dll x:\path\do\dllname.dll
//		func funcName type typeName
//
//		n:						// Specifies the next type begins at offset n
//
//		type;					// Placeholder
//		type[n];				// Placeholder
//		type name;				// Single data element
//		type name[n];			// Multiple data elements
//		type* name;				// Pointer to the indicated type
//
// Note:  type can be a previously defined struct
//
//////
	struct cdxHead
	|dll cdx.dll
	|	func cdx_decode_keyfor		type keyfor
	{

0:		u32		root;
		u32		free;
12:		u16		keyLength;
		u8		options;
		u8		signature;
502:	u16		order;
506:	u16		forClauseLength;

510:	keyfor	"Key and FOR Clause:";
	};
